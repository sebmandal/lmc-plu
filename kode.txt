START			INP
			STA PLU
			BRZ END
			BRA GETCOUNT


GETCOUNT		INP
			STA COUNT

			BRP SETPRICE
			BRA GETCOUNT


SETPRICE		LDA PLU
			SUB 405
			BRZ SET405

			LDA PLU
			SUB 307
			BRZ SET307

			LDA PLU
			SUB 303
			BRZ SET303

			LDA ZER
			STA TOTAL
			BRA END


SET405			LDA P405
			STA PRICE
			BRA MULT


SET307			LDA P307
			STA PRICE
			BRA MULT


SET303			LDA P303
			STA PRICE
			BRA MULT


MULT			LDA COUNT
			BRZ START

			LDA TOTAL
			ADD PRICE
			STA TOTAL

			LDA COUNT
			SUB ONE
			STA COUNT

			BRA MULT


END			LDA TOTAL
			OUT


PLU			DAT 0
PRICE			DAT 0
COUNT			DAT 0
TOTAL			DAT 0


P405			DAT 5
P307			DAT 7
P303			DAT 9


ZER			DAT 0
ONE			DAT 1
405			DAT 405
307			DAT 307
303			DAT 303

IFI_BRUKER		DAT swmandal
