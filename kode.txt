START 		INP
		BRZ OUTPUT
		STA PLU
		LDA PLU
		SUB PLU1
		BRZ SETPRICE1
		LDA PLU
		SUB PLU2
		BRZ SETPRICE2
		BRA START

SETPRICE1	LDA PRICE1
		STA PRICE
		BRA GETQTY

SETPRICE2 	LDA PRICE2
		STA PRICE
		BRA GETQTY

GETQTY 		INP
		STA QTY
		LDA ZERO
		STA SUM
	
LOOP 		LDA QTY
		BRZ ADDTOT
		LDA PRICE
		ADD SUM
		STA SUM
		LDA QTY
		SUB ONE
		STA QTY
		BRA LOOP

ADDTOT 		LDA TOTAL
		ADD SUM
		STA TOTAL
		BRA START

OUTPUT 		LDA TOTAL
		OUT

		// printe navnet mitt lol
		LDA NEWLINE
		OTC
		LDA LETTER1
		OTC
		LDA LETTER2
		OTC
		LDA LETTER3
		OTC
		LDA LETTER4
		OTC
		LDA LETTER5
		OTC
		LDA LETTER6
		OTC
		LDA LETTER7
		OTC
		LDA LETTER8
		OTC

		HLT

PLU		DAT 0
PLU1		DAT 1
PLU2		DAT 2
PRICE1		DAT 100
PRICE2		DAT 200
PRICE		DAT 0
QTY		DAT 0
SUM		DAT 0
TOTAL		DAT 0

ONE		DAT 1
